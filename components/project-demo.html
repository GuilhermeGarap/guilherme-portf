Service Layer
@Service @Transactional public class EstudanteService { public Estudante criar(EstudanteDTO dto) { if (repository.existsByMatricula(dto.getMatricula())) { throw new MatriculaDuplicadaException(); } Estudante estudante = new Estudante(); estudante.setNome(dto.getNome()); estudante.setMatricula(dto.getMatricula()); estudante.setEmail(dto.getEmail()); estudante.setCurso(dto.getCurso()); estudante.setDataNascimento(dto.getDataNascimento()); return repository.save(estudante); } public Estudante buscarPorId(Long id) { return repository.findById(id) .orElseThrow(() -> new EstudanteNotFoundException()); } }
