@Service @Transactional public class ReservaService { public Reserva criarReserva(ReservaDTO dto) { // Validação de disponibilidade do carro no período if (!carroService.verificarDisponibilidade( dto.getCarroId(), dto.getDataInicio(), dto.getDataFim())) { throw new CarroIndisponivelException(); } // Cálculo automático do preço total da reserva BigDecimal precoTotal = calcularPrecoTotal( dto.getDataInicio(), dto.getDataFim(), dto.getCarroId()); // Criação e configuração da nova reserva Reserva reserva = new Reserva(); reserva.setCarro(carroService.buscarPorId(dto.getCarroId())); reserva.setCliente(clienteService.buscarPorId(dto.getClienteId())); reserva.setDataInicio(dto.getDataInicio()); reserva.setDataFim(dto.getDataFim()); reserva.setPrecoTotal(precoTotal); // Persistência da reserva no banco de dados return reservaRepository.save(reserva); } }

API REST - Controller
@RestController @RequestMapping("/api/produtos") @Validated public class ProdutoController { @PostMapping public ResponseEntity<ProdutoDTO> cadastrarProduto( @Valid @RequestBody ProdutoDTO produtoDTO) { Produto produto = produtoService.cadastrarProduto(produtoDTO); return ResponseEntity.status(HttpStatus.CREATED) .body(ProdutoDTO.fromEntity(produto)); } @GetMapping public ResponseEntity<List<ProdutoDTO>> listarProdutos() { List<Produto> produtos = produtoService.listarProdutos(); List<ProdutoDTO> produtosDTO = produtos.stream() .map(ProdutoDTO::fromEntity) .collect(Collectors.toList()); return ResponseEntity.ok(produtosDTO); } @PutMapping("/{id}/estoque") public ResponseEntity<void> atualizarEstoque( @PathVariable Long id, @RequestParam Integer quantidade) { produtoService.atualizarEstoque(id, quantidade); return ResponseEntity.noContent().build(); } @DeleteMapping("/{id}") public ResponseEntity<void> deletarProduto(@PathVariable Long id) { produtoService.deletarProduto(id); return ResponseEntity.noContent().build(); } }
Modelo de Dados - JPA
@Entity @Table(name = "produtos") @Data @NoArgsConstructor @AllArgsConstructor @Builder public class Produto { @Id @GeneratedValue(strategy = GenerationType.IDENTITY) private Long id; @Column(nullable = false, unique = true) private String codigoBarras; @Column(nullable = false) private String nome; @Column(columnDefinition = "TEXT") private String descricao; @Column(nullable = false, precision = 10, scale = 2) private BigDecimal preco; @Column(nullable = false) private Integer quantidadeEstoque; @Column(nullable = false) private LocalDateTime dataCadastro; @PrePersist protected void onCreate() { dataCadastro = LocalDateTime.now(); } @OneToMany(mappedBy = "produto", cascade = CascadeType.ALL) private List<Venda> vendas; }

// Controller REST para gerenciar operações de estudantes @RestController @RequestMapping("/api/estudantes") public class EstudanteController { // Endpoint para criar um novo estudante @PostMapping public ResponseEntity<Estudante> criarEstudante( @Valid @RequestBody EstudanteDTO dto) { // Chama o service para criar o estudante Estudante estudante = service.criar(dto); // Retorna status 201 (Created) com o estudante criado return ResponseEntity.status(201).body(estudante); } // Endpoint para buscar estudante por ID @GetMapping("/{id}") public ResponseEntity<Estudante> buscarPorId(@PathVariable Long id) { // Busca o estudante no service Estudante estudante = service.buscarPorId(id); // Retorna status 200 (OK) com o estudante encontrado return ResponseEntity.ok(estudante); } // Endpoint para atualizar dados do estudante @PutMapping("/{id}") public ResponseEntity<Estudante> atualizar( @PathVariable Long id, @Valid @RequestBody EstudanteDTO dto) { // Atualiza o estudante no service Estudante estudante = service.atualizar(id, dto); // Retorna status 200 (OK) com o estudante atualizado return ResponseEntity.ok(estudante); } }
Service Layer
// Service layer para lógica de negócio dos estudantes @Service @Transactional public class EstudanteService { // Método para criar um novo estudante public Estudante criar(EstudanteDTO dto) { // Validação de matrícula única if (repository.existsByMatricula(dto.getMatricula())) { throw new MatriculaDuplicadaException(); } // Criação do novo estudante Estudante estudante = new Estudante(); // Mapeamento dos dados do DTO para a entidade estudante.setNome(dto.getNome()); estudante.setMatricula(dto.getMatricula()); estudante.setEmail(dto.getEmail()); estudante.setCurso(dto.getCurso()); estudante.setDataNascimento(dto.getDataNascimento()); // Salva e retorna o estudante criado return repository.save(estudante); } // Método para buscar estudante por ID public Estudante buscarPorId(Long id) { // Busca no repository e lança uma exceção se não encontrar return repository.findById(id) .orElseThrow(() -> new EstudanteNotFoundException()); } }
Entity Model
// Entidade JPA que representa a tabela de estudantes @Entity @Table(name = "estudantes") public class Estudante { // Chave primária auto-incrementada @Id @GeneratedValue(strategy = GenerationType.IDENTITY) private Long id; // Nome do estudante (obrigatório) @Column(nullable = false) private String nome; // Matrícula única do estudante @Column(unique = true, nullable = false) private String matricula; // Email do estudante (obrigatório) @Column(nullable = false) private String email; // Curso do estudante (obrigatório) @Column(nullable = false) private String curso; // Data de nascimento do estudante @Column(name = "data_nascimento") private LocalDate dataNascimento; // Relacionamento One-to-Many com carteirinhas @OneToMany(mappedBy = "estudante", cascade = CascadeType.ALL) private List<Carteirinha> carteirinhas; // Getters e Setters }
Validações
// DTO com validações Bean Validation para entrada de dados public class EstudanteDTO { // Validação de nome obrigatório com tamanho específico @NotBlank(message = "Nome é obrigatório") @Size(min = 2, max = 100, message = "Nome deve ter entre 2 e 100 caracteres") private String nome; // Validação de matrícula com regex para 8 dígitos @NotBlank(message = "Matrícula é obrigatória") @Pattern(regexp = "^[0-9]{8}$", message = "Matrícula deve ter 8 dígitos") private String matricula; // Validação de email com formato válido @NotBlank(message = "Email é obrigatório") @Email(message = "Email deve ser válido") private String email; // Validação de curso obrigatório @NotBlank(message = "Curso é obrigatório") private String curso; // Validação de data de nascimento no passado @NotNull(message = "Data de nascimento é obrigatória") @Past(message = "Data de nascimento deve ser no passado") private LocalDate dataNascimento; // Getters e Setters }
